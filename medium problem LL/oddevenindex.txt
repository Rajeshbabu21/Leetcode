/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode oddEvenList(ListNode head) {
        ArrayList<Integer> ans = new ArrayList<>();
        ListNode temp = head;

        if(head == null)
        {
            return null;
        }

        // this next != null is for the last it is standing in the last amake the false
        while(temp != null && temp.next != null)
        {
            ans.add(temp.val);
            temp = temp.next.next;
        }
        if(temp != null)
        {
            ans.add(temp.val);
        }

        temp = head.next;

        // this next != null is for the last it is standing in the last amake the false

        while(temp != null && temp.next != null)
        {
            ans.add(temp.val);
            temp = temp.next.next;
        }

        if(temp != null)
        {
            ans.add(temp.val);
        }
        temp = head;
        int index = 0;

        while(temp != null)
        {
            temp.val = ans.get(index);
            index++;
            temp = temp.next;
        }

        return head;
        
        
    }
}




/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode oddEvenList(ListNode head) {

        if (head == null || head.next == null) {
            return head;
        }

        ListNode odd = head;
        ListNode even = head.next;
        ListNode evenHead = even;

        while (even != null && even.next != null) {

            odd.next = odd.next.next;
            odd = odd.next;

            even.next = eve.next.next;
            even = even.next;
        }

        odd.next = evenHead;

        return head;
    }
