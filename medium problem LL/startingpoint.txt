/**
 * Definition for singly-linked list.
 * class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode detectCycle(ListNode head) {
        Map<ListNode,Integer> ans = new HashMap<>();
        ListNode temp = head;

        if(head == null || head.next == null)
        {
            return null;
        }

        while(temp != null)
        {
            if(ans.containsKey(temp))
            {
                return temp;
            }
            else{
                ans.put(temp,1);
            }
            temp = temp.next;
        }
        return null;


        
        
    }
}
