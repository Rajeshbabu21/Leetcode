/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) {
 *         val = x;
 *         next = null;
 *     }
 * }
 */
public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        
        Map<ListNode,Integer> ans = new HashMap<>();
        ListNode temp = headA;

        while(temp != null)
        {
            ans.put(temp,1);
            temp = temp.next;
        }
        temp = headB;
        while(temp != null)
        {
            if(ans.containsKey(temp))
            {
                return temp;
            }
            else{
                ans.put(temp,1);

            }
            temp = temp.next;
        }
        return null;


        
    }
}
