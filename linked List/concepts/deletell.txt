// deletion of position k
class Solution {
    public void deleteNode(Node head,int k) {


// for one and zer oelement 
        if(head == null)
        {
            return null;
        }

// this is head
        if(k == 1)
        {
            Node temp = head;
            head = head.next;
            temp.next = null;
            return head;
        }
        
        int count = 0;
        Node temp = head;
        Node prev  = head;
        // to reach the last element
        while(temp!= null)
        {
            count++;
            if(count == k)
            {
                prev.next = temp.next;
            }
            prev = temp;
            temp = temp.next;
        }

        return head;

        
    }
}
