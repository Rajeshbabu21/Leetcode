class Solution {
    public void Insert(Node node,int el,int x) {
        if(head == null)
        {
            
            return null;
        }

        if(head.data == x)
        {
            Node temp = new Node(el);
            temp.next = head;
            return temp;
        }

        Node temp = head;
        Node prev  = head;

        // for this we nned to till the end 
        while(temp != null)
        {

            if(temp.data == x )
            {
                Node nn = new Node(el);
                nn.next = temp;
                prev.next = nn;
                break;
            }
            prev = temp;
        temp = temp.next;
        return head;
        }
        
    }
}
