class Solution {
    public int maxDepth(String s) {
        int n = s.length();
        int deeper = 0;
        int count = 0;

        for(int i=0;i<n;i++)
        {

            if(s.charAt(i)=='(')
            {
                count++;
                deeper = Math.max(count,deeper);
            }
            else if(s.charAt(i)==')'){
                count--;
            }
        }

        return deeper;



        
    }
}
