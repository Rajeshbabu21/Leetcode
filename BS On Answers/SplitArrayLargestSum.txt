Given an integer array nums and an integer k, split nums into k non-empty subarrays such that the largest sum of any subarray is minimized.

Return the minimized largest sum of the split.

A subarray is a contiguous part of the array.

 

Example 1:

Input: nums = [7,2,5,10,8], k = 2
Output: 18
Explanation: There are four ways to split nums into two subarrays.
The best way is to split it into [7,2,5] and [10,8], where the largest sum among the two subarrays is only 18.



class Solution {
    public int maxfunction(int[] nums)
    {
        
        int max = Integer.MIN_VALUE;
        for(int i=0;i<nums.length;i++)
        {
            if(nums[i] > max)
            {
                max= nums[i];
            }
        }

        return max;
    }

    public int sumfunction(int[] nums)
    {
        int sum = 0;
        for(int arr:nums)
        {
            sum +=arr;
        }
        return sum;
    }

    public int findthelargestsum(int[] nums,int mid)
    {
        int split = 1;
        int sum = 0;

        for(int i = 0;i<nums.length;i++)
        {
            if(sum + nums[i] > mid)
            {
                split+=1;
                sum = nums[i];

            }
            else{
                sum += nums[i];
            }
        }

        return split;
    }
    public int splitArray(int[] nums, int k) {
        int low = maxfunction(nums);
        int high = sumfunction(nums);
        int ans = -1;

        while(low <= high)
        {
            int mid = (low + high) / 2;
            int findthelargestsum = findthelargestsum(nums,mid); 

            if(findthelargestsum <= k)
            {
                ans = mid;
                high=mid-1;
            }
            else{
                low = mid +1;
            }
            
        }


return ans;
        
    }
}
