// low will always points to the not possible answer
// high will points to the possible answer
// vice versa 


class Solution {
    public int findsqrt(int n) {
        int low = 1;
        int high = n;
        ans= -1;

        while(low <= high)
        {
            int mid = (low + high) / 2;
            
            if(mid * mid <= n)
            {
                ans = mid;

                // we need highest
                low = mid +1;
            }

            // right
            else{

                // if if element is greater
                high = mid-1;
                
        }

        return ans or high;


    }
    }
}
