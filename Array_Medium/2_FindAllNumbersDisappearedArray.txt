class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        int n = nums.length;

        HashMap<Integer,Integer> map = new HashMap<>();
        ArrayList<Integer> ans = new ArrayList<>();

        for(int i : nums)
        {
            map.put(i,1);
        }

        for(int i=1;i<=n;i++)
        {
            if(!map.containsKey(i))
            {
                ans.add(i);
            }
        }

        return ans;

        
    }
}
