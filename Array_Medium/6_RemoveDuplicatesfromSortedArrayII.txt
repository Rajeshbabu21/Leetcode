class Solution {
    public int removeDuplicates(int[] nums) {
        int n = nums.length;
        int index = 0;
        LinkedHashMap<Integer,Integer> ans = new LinkedHashMap<>();

        for(int i : nums){
            ans.put(i,ans.getOrDefault(i,0)+1);
        }

        ArrayList<Integer> list = new ArrayList<>(ans.keySet());

        for(int j : list){
            int freq = ans.get(j);
            int times = Math.min(freq, 2);
            for(int i=0;i<times;i++)
            {
                nums[index] = j;
                index++;
            }
        }

        return index;
        
    }
}
