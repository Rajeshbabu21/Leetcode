class Solution {
    public String reverseOnlyLetters(String s) {
        int n = s.length();
        char[] ans = s.toCharArray();
        int left = 0;
        int right  = n-1;

        while(left < right)
        {
            while(left < right && !Character.isLetter(ans[left]))
            {
                left++;
            }

            while(left < right && !Character.isLetter(ans[right]))
            {
                right--;
            }

            char temp = ans[left];
            ans[left] = ans[right];
            ans[right] = temp;

            left++;
            right--;
        }

        return new String(ans);
        
    }
}
