class Solution {
    public int[] runningSum(int[] nums) {

        int n = nums.length;
        ArrayList<Integer> ans = new ArrayList<>();

        for(int i=0;i<n;i++)
        {
            int sum = 0;
            for(int j = 0;j<=i;j++)
            {
                sum += nums[j];
            }
            ans.add(sum);
            

        }
        int i =0;
        for(int j : ans){
            nums[i] = j;
            i++;
        }

        return nums;

        
    }
}
