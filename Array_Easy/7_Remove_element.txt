class Solution {
    public int removeElement(int[] nums, int val) {
        int n = nums.length;
        int count = 0;
        int index = 0;
        ArrayList<Integer> ans = new ArrayList<>();

        for(int num:nums){
            if(num != val)
            {
                ans.add(num);
            }
        }

        for(int num:ans){
            nums[index] = num;
            index++;
        }

        return ans.size();
    }
}
