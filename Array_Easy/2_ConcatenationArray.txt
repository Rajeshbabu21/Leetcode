class Solution {
    public int[] getConcatenation(int[] nums) {

        int n = nums.length;
        int n2 = 2*n;
        int[] array = new int[n2];

        for(int i=0;i<n;i++)
        {
            array[i] = nums[i];
        }
        int index = n; 

        for(int i=0;i<n;i++)
        {
            array[index] = nums[i];
            index++;
        }

        return array;





        
    }
}
