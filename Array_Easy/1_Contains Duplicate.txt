class Solution {
    public boolean containsDuplicate(int[] nums) {
        HashMap<Integer,Integer> ans = new HashMap<>();
        int n = nums.length;

        for(int i=0;i<n;i++)
        {
            if(ans.containsKey(nums[i]))
            {
                return true;
            }
            else{
                ans.put(nums[i],1);
            }
        }

        return false;
        
    }
}
