class Solution {
    public boolean check(int[] nums) {
        int n = nums.length;
        int bre = 0;

        for(int i=0;i<n-1;i++)
        {
            if(nums[i]> nums[i+1])
            {
                bre +=1;
            }

        }
        if(nums[n-1] > nums[0])
        {
            bre +=1;
        }

        return bre <=1;
        
    }
}
