class Solution {
    public int findDuplicate(int[] nums) {
        int n = nums.length;
        int ans = nums[0];

        for(int i=0;i<n;i++)
        {
            int count = 0;
            for(int j=0;j<n;j++)
            {
                if(nums[i] == nums[j])
                {
                    count = count+1;
                }
            }

            if(count > 1)
            {
                ans = nums[i];
            }
        }

        return ans;

        
        
    }
}
