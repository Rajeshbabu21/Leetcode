class Solution {
    public boolean containsDuplicate(int[] nums) {

        HashMap<Integer,Integer> ans = new HashMap<>();

        int n = nums.length;

        for(int a : nums)
        {
            if(ans.containsKey(a))
            {
                return true;
            }
            else{
                ans.put(a,1);
            }
        }

        return false;
        
    }
}
