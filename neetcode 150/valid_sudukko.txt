import java.util.HashMap;

class Solution {
    public boolean isValidSudoku(char[][] board) {

        // Check rows
        for (int i = 0; i < 9; i++) {
            HashMap<Character, Integer> map = new HashMap<>();

            for (int j = 0; j < 9; j++) {
                if (board[i][j] == '.') {
                    continue;
                }

                if (map.containsKey(board[i][j])) {
                    return false;
                }

                map.put(board[i][j], 1);
            }
        }

        // Check columns
        for (int j = 0; j < 9; j++) {
            HashMap<Character, Integer> map = new HashMap<>();

            for (int i = 0; i < 9; i++) {
                if (board[i][j] == '.') {
                    continue;
                }

                if (map.containsKey(board[i][j])) {
                    return false;
                }

                map.put(board[i][j], 1);
            }
        }

        // Check 3×3 boxes
        for (int row = 0; row < 9; row += 3) {
            for (int col = 0; col < 9; col += 3) {

                HashMap<Character, Integer> map = new HashMap<>();

                for (int i = row; i < row + 3; i++) {
                    for (int j = col; j < col + 3; j++) {

                        if (board[i][j] == '.') {
                            continue;
                        }

                        if (map.containsKey(board[i][j])) {
                            return false;
                        }

                        map.put(board[i][j], 1);
                    }
                }
            }
        }

        return true;
    }
}
