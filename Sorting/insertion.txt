class Solution {
    public void Insertion_sort(int[] arr) {

        int n = arr.length;

        for(int i=0;i<n;i++)
        {
            int j = i;
            while(j > 0 && arr[j-1] > arr[j])
            // 6 5 so swap
            {
                int temp = arr[j];
                arr[j] = arr[j-1];
                arr[j-1] = temp
                j--;
            }
        }

           
    }
}
