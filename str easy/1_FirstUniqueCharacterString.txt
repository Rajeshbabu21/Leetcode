class Solution {
    public int firstUniqChar(String s) {
        HashMap<Character,Integer> ans = new HashMap<>();
        int n = s.length();
        for(int i=0;i<n;i++)
        {
            char a = s.charAt(i);
            ans.put(a,ans.getOrDefault(a,0)+1);
        }

        for(int i=0;i<n;i++)
        {
            if(ans.get(s.charAt(i)) == 1)
            {
                return i;
            }
        }

        return -1;

        
        
    }
}
